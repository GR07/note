
# 取差集
```js
// set 实现
let a = new Set([1, 2, 3])
let b = new Set([4, 2, 3])
let differ = Array.from(new Set([...a].filter(item => !b.has(item)))) // [1]


// includes 实现
arr1 = [1, 2, 3, 4];
arr2 = [3, 1];
let difference = arr1.filter(x => !arr2.includes(x)) // [2, 4]
```


# 取交集
```js
// set 实现
let a = new Set([1, 2, 3])
let b = new Set([4, 2, 3])
let inter = [...new Set([...a].filter(item => b.has(item)))] // [2, 3]
```


# 数组合并去重
```js
// set 实现
let a = new Set([1, 2, 3])
let b = new Set([4, 2, 3])
let union = [...new Set([...a, ...b])] // [1, 2, 3, 4]


// Lodash _.uniqWith() 实现（可以处理引用类型）
var objects = [{ 'x': 1, 'y': 2 }, { 'x': 2, 'y': 1 }, { 'x': 1, 'y': 2 }];
_.uniqWith(objects, _.isEqual);
// _.isEqual(value,other)用于执行深比较来确定两者的值是否相等。 _.uniqWith()做去重处理。
```

# 数组对象 转 对象
```js
// reduce 实现
const arr = [
    { key: 'CN', name: 'China' },
    { key: 'US', name: 'USA' },
    { key: 'JP', name: 'Japan' },
    { key: 'EU', name: 'Eurozone' }
]

const obj = arr.reduce((acc, cur) => {
    acc[cur.key] = cur.name
    return acc
}, {})
```
