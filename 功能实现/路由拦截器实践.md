# 利用拦截器结合令牌做数据持久化

```js
import router from './router'
import store from './store'
import { getToken } from './utils/auth.js'
const whiteList = ['/', '/login', '/xxx']
router.beforeEach(async (to, from, next) => {
  if (to.meta.title) {
    document.title = to.meta.title
  }
  const hasToken = getToken()
  if (hasToken) {
    const hasId = store.getters.id
    if (hasId || whiteList.includes(to.path)) {
      next()
    } else {
      try {
        await store.dispatch('info/getRuleAndDetail')
        next({ ...to, replace: true })
      } catch (err) {
        next('/login')
      }
    }
  } else {
    if (whiteList.includes(to.path)) {
      next()
    } else {
      next('/login')
    }
  }
})
```

# store 数据获取&处理

```js
const actions = {

  getRuleAndDetail ({ commit, dispatch }) {
    return new Promise((resolve, reject) => {
      Promise.all(createPromiseList()).then(res => {

        const [rule, detail] = res

        const configList = configToArray(rule)

        const filterShowList = configList.filter(item => Object.is(DISPLAY_STATUS.TRUE, +item.displayFlag))

        const list = mergeOptions(filterShowList, detail)

        commit('SET_FORM_CONFIG', list)

        commit('SET_ALL_OCR_LIST', detail.certificateList)

        commit('SET_ALL_FILE', detail.certificateList)

        commit('SET_ID', detail)

        commit('SET_SELF_OWNED', detail)

        commit('SET_HIRE', detail)

        dispatch('getDic')

        dispatch('progress/setAll', list, { root: true })

        resolve(detail)
      }).catch(({ msg }) => {
        reject(msg)
      })
    })
  },

  async getDic ({ state, commit }) {
    try {

      const dynamicDic = state.formConfig.flatMap(item => item.relationList).map(item => item.dictionary)

      const uniqueArray = [...new Set([...dynamicDic, ...CONSTANT_DIC])].filter(item => /_+/.test(item + ''))

      const res = await http('xxx', uniqueArray)

      const map = res.reduce((acc, cur) => {
        acc[cur.code] = cur.childrenLookupValues.map(({ lookupValue, lookupCode }) => ({ label: lookupValue, value: lookupCode }))
        return acc
      }, {})

      commit('SET_DIC', map)

    } catch (err) {
      console.log(`🚀 ~ getDic ~ err:`, err)
    }
  },

  logout () {
    return new Promise((resolve) => {
      setToken('')
      resetRouter()
      resolve()
    })
  }

}
```
